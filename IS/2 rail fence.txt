from __future__ import print_function
S = "SAEKONDHWA"
D = 3

grid =[["" for i in range(len(S))] for j in range(D)]

flag = 0
row = 0

for i in range(len(S)):
    grid[row][i]=S[i]
    
    if row == 0:
        flag=0
    elif row == D-1:
        flag = 1
        
    if flag == 0:
        row += 1
    else:
        row -= 1
    
for i in grid:
    for j in i:
        print(j, end = "")
}
